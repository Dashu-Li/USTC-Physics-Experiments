\section{实验题目}
$\beta$-$\gamma$符合法测量放射源绝对强度

\section{实验目的}
\begin{enumerate}
    \item 掌握符合测量系统的延迟调整方法，理解符合曲线与时间分辨的物理意义。
    \item 通过真实符合曲线的高斯拟合，估计系统符合时间分辨宽度。
    \item 用 $\beta$-$\gamma$ 符合法测定 $^{60}\mathrm{Co}$ 放射源的绝对强度（活度）。
\end{enumerate}

\section{实验原理}
\subsection{符合分辨时间}
符合电路存在一定分辨时间。若两路脉冲到达时间差小于该时间窗口，系统将其判作“同时到达”，并产生符合输出。该最小可分辨时间间隔记为 $\tau$。

\subsection{符合分辨时间的测量方法}
\subsubsection{偶然符合法}
设两道平均计数率分别为 $n_1$、$n_2$，当两道输出脉冲宽度均为 $\tau$ 时，偶然符合计数率为
\begin{equation}
    n_{\mathrm{rc}} = 2\tau n_1 n_2.
\end{equation}
考虑本底符合计数率 $n_{\mathrm{b}12}$ 后，有
\begin{equation}
    n'_{\mathrm{rc}} = 2\tau n_1 n_2 + n_{\mathrm{b}12}.
\end{equation}
因此作图 $n'_{\mathrm{rc}}-n_1n_2$ 可由斜率求得 $\tau$。

\subsubsection{瞬时符合曲线法}
在符合系统中改变两路相对延迟时间 $t$，符合计数率将随 $t$ 形成一条峰形分布曲线。若使用探测器真实信号，由于脉冲前沿存在统计涨落，曲线近似钟形，其半高全宽可表征系统时间分辨能力。

\subsection{$\beta$-$\gamma$符合法测放射源活度}
对于 $^{60}\mathrm{Co}$ 源，若源活度为 $A$，$\beta$、$\gamma$ 道探测效率分别为 $p_{\beta}$、$p_{\gamma}$，则有
\begin{align}
    n_{\beta} &= A p_{\beta}, \\
    n_{\gamma} &= A p_{\gamma}, \\
    n_{\beta\gamma} &= A p_{\beta} p_{\gamma}.
\end{align}
故理想情形下
\begin{equation}
    A = \frac{n_{\beta} n_{\gamma}}{n_{\beta\gamma}}.
\end{equation}

考虑本底、$\gamma$ 射线串入 $\beta$ 道以及偶然符合修正后，按实验讲义数据处理关系可写为
\begin{equation}
\begin{aligned}
n_{\beta\gamma}(\gamma,\gamma) &= n_{\beta\gamma}(\gamma+b)-n_{\beta\gamma}(b)-2\tau_{\mathrm{e}} n_{\beta}(\gamma+b)n_{\gamma}(\gamma+b), \\
n_{\beta\gamma}(\beta,\gamma) &= n_{\beta\gamma}(\beta+\gamma+b)-2\tau_{\mathrm{e}} n_{\beta}(\beta+\gamma+b)n_{\gamma}(\gamma+b)-n_{\beta\gamma}(b)-n_{\beta\gamma}(\gamma,\gamma), \\
A &= \frac{\left[n_{\beta}(\beta+\gamma+b)-n_{\beta}(\gamma+b)\right]\left[n_{\gamma}(\gamma+b)-n_{\gamma}(b)\right]}{n_{\beta\gamma}(\beta,\gamma)}.
\end{aligned}
\label{eq:A}
\end{equation}
其中 $\tau_{\mathrm{e}}$ 为电子学符合分辨时间（用于偶然符合修正，单位为秒）。

真实符合曲线采用高斯模型拟合：
\begin{equation}
    N(t)=N_0+N_p\exp\!\left[-\frac{{(t-\mu)}^2}{2\sigma^2}\right],
\end{equation}
其半高全宽
\begin{equation}
    \mathrm{FWHM}=2\sqrt{2\ln 2}\,\sigma.
\end{equation}

\section{实验过程}
\begin{enumerate}
    \item 按 $\beta$-$\gamma$ 符合装置框图连接系统（见图~\ref{fig:setup-scheme}）。
    \item 用精密脉冲发生器作为输入，调节放大器增益、成形时间、单道阈值和符合延迟，使符合通道稳定输出。
    \item 固定一路延迟，连续调节另一路延迟，在示波器上观察符合输出波形，并在有符合输出的延迟范围内选取若干点测量符合计数。
    \item 以探测器真实信号替代标准脉冲，按 \qty{0.1}{\micro\second} 步进扫描，测量瞬时符合曲线各点计数。
    \item 将延迟固定在符合曲线中部附近，按三种工况（正常、用铝盖住、移除放射源）分别测量 $\beta$ 道、$\gamma$ 道与符合道计数，并按修正式计算活度。
\end{enumerate}

\subsection{第二组高斯拟合与符合时间分辨}
将第二组计数 $N(t)$ 按高斯模型拟合，得到
\begin{align}
    N_p &= 83.90, \\
    \mu &= \SI{4.786}{\micro\second}, \\
    \sigma &= \SI{0.818}{\micro\second}, \\
    N_0 &= -0.66 \approx 0.
\end{align}
由此
\begin{equation}
    \mathrm{FWHM}=2\sqrt{2\ln2}\,\sigma=\SI{1.926}{\micro\second}.
\end{equation}
按讲义图示定义（半高宽标为 $2\tau$），有
\begin{equation}
    2\tau = \mathrm{FWHM},\qquad
    \tau = \frac{\mathrm{FWHM}}{2}=\SI{0.9632}{\micro\second}.
\end{equation}
下文偶然符合修正项按该定义使用 $\tau=\SI{0.9625}{\micro\second}$。

\subsection{第三组绝对强度计算}
先将第三组计数换算为计数率（单位：\si{\per\second}）：
\begin{align}
    n_{\beta}(\beta+\gamma+b) &= \frac{8632}{300}=28.7733, \\
    n_{\beta}(\gamma+b) &= \frac{2630}{300}=8.7667, \\
    n_{\gamma}(\gamma+b) &= \frac{120312}{300}=401.0400, \\
    n_{\gamma}(b) &= \frac{51469}{300}=171.5633, \\
    n_{\beta\gamma}(\beta+\gamma+b) &= \frac{388}{300}=1.2933, \\
    n_{\beta\gamma}(\gamma+b) &= \frac{134}{300}=0.4467, \\
    n_{\beta\gamma}(b) &= \frac{8}{300}=0.0267.
\end{align}

于是
\begin{align}
    n_{\beta} &= n_{\beta}(\beta+\gamma+b)-n_{\beta}(\gamma+b)
    =20.0067, \\
    n_{\gamma} &= n_{\gamma}(\gamma+b)-n_{\gamma}(b)
    =229.4767.
\end{align}
其中 $n_{\beta}(\gamma+b)$ 为本底与 $\gamma$ 串入 $\beta$ 道的合量，按本实验测量条件不再拆分为 $n_{\beta}(b)$ 与 $n_{\beta}(\gamma)$ 两项。

按实验要求固定延迟 $t=\SI{4.3}{\micro\second}$，并按上式取 $\tau=\SI{0.9632}{\micro\second}$ 代入式~\eqref{eq:A}：
\begin{align}
    n_{\beta\gamma}(\gamma,\gamma)
    &=0.4467-0.0267-2\times 0.9632\times10^{-6}\times 8.7667\times 401.0400 \\
    &=0.41323, \\
    n_{\beta\gamma}(\beta,\gamma)
    &=1.2933-2\times 0.9632\times10^{-6}\times 28.7733\times 401.0400-0.0267-0.41323 \\
    &=0.83121, \\
    A
    &=\frac{20.0067\times229.4767}{0.83121}
    =\SI{5.52e3}{\becquerel}.
\end{align}

故本次测得 $^{60}\mathrm{Co}$ 源活度约为
\begin{equation}
    A\approx \SI{5.52}{\kilo\becquerel}.
\end{equation}

若仅按泊松统计并忽略系统误差，近似相对标准不确定度可估算为
\begin{equation}
    u_r(A)\approx\sqrt{\frac{1}{N_{\beta}}+\frac{1}{N_{\gamma}}+\frac{1}{N_{\beta\gamma}}}\approx 6.52\%.
\end{equation}

\section{结果与讨论}
\begin{enumerate}
    \item 第一组标准信号数据给出了符合输出的延迟平台特征，与实验过程中“观察符合输出并选点测量”的步骤一致。
    \item 第二组真实符合曲线高斯拟合峰位在 $\mu\approx\SI{4.79}{\micro\second}$，半高全宽约 \SI{1.93}{\micro\second}。
    \item 在固定延迟 \SI{4.3}{\micro\second} 的测量条件下，按讲义图示 $2\tau$ 定义并显式计算 $n_{\beta\gamma}(\gamma,\gamma)$、$n_{\beta\gamma}(b)$ 后，得到 $^{60}\mathrm{Co}$ 源绝对强度约 \SI{5.52}{\kilo\becquerel}。
    \item 误差主要来自：统计涨落（尤其符合道计数较低）、阈值与增益漂移、延迟设置偏离峰位、以及本底修正与偶然符合修正模型近似。
    \item \textbf{与参考值（1000$\sim$3000\,\unit{\becquerel}）的偏差分析}：本次计算严格遵循讲义给出公式演化，实际求得活度约 \SI{5.52}{\kilo\becquerel}。偏高原因可能为：一是放射源的新旧更替，$^{60}\mathrm{Co}$ 半衰期为 5.27 年，新购入的源活度自然远超老旧源的参考值；二是 $^{60}\mathrm{Co}$ 各发生一次 $\beta$ 衰变会产生 2 个级联 $\gamma$ 光子，若部分讲义参考数据是按引入 $\frac{1}{2}$ 修正因子的公式计算得出的，则本次结果折半后约为 \SI{2.76}{\kilo\becquerel}，恰好落于参考区间内。
\end{enumerate}
